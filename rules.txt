local function getScene() return PlayScene.Value end

local function resolve(path)
	local scene=getScene()
	if not scene or not path or path=="" then return nil end
	if typeof(path)=="Instance" then return path end
	path=tostring(path)
	if path:find("%.") then
		local obj=scene
		for part in path:gmatch("[^%.]+") do
			if not obj then return nil end
			obj=obj:FindFirstChild(part)
		end
		return obj
	end
	return scene:FindFirstChild(path,true)
end

local function resolveWithRetry(path, maxRetries, retryDelay)
	maxRetries = maxRetries or 20
	retryDelay = retryDelay or 0.15
	for attempt = 1, maxRetries do
		local obj = resolve(path)
		if obj then
			if attempt > 1 then logFire("[FS] '"..tostring(path).."' nach "..attempt.." Versuchen") end
			return obj
		end
		if attempt < maxRetries then task.wait(retryDelay) end
	end
	logFire("[FS] '"..tostring(path).."' nicht gefunden nach "..maxRetries)
	return nil
end

local function j(v)
	local t=typeof(v)
	if t=="boolean" then return v and "true" or "false" end
	if t=="number"  then return tostring(v) end
	if t=="string"  then return '"'..v:gsub('\\','\\\\'):gsub('"','\\"'):gsub("\n","\\n")..'"' end
	if t=="Color3"  then return string.format('"#%02X%02X%02X"',math.floor(v.R*255),math.floor(v.G*255),math.floor(v.B*255)) end
	if t=="UDim2"   then return string.format('{"xs":%.3f,"xo":%d,"ys":%.3f,"yo":%d}',v.X.Scale,v.X.Offset,v.Y.Scale,v.Y.Offset) end
	if t=="UDim"    then return string.format('{"s":%.3f,"o":%d}',v.Scale,v.Offset) end
	if t=="Vector2" then return string.format('{"x":%.2f,"y":%.2f}',v.X,v.Y) end
	if t=="EnumItem" then return '"'..tostring(v)..'"' end
	return '"'..tostring(v):gsub('"','\\"')..'"'
end
local function serInst(inst,d)
	d=d or 0; if d>8 then return '{"_t":true}' end
	local p={'"Name":'..j(inst.Name),'"Class":"'..inst.ClassName..'"'}
	local ok,ig=pcall(function() return inst:IsA("GuiObject") end)
	if ok and ig then for _,prop in ipairs(SAFE_PROPS.GuiObject) do local ok2,v=pcall(function() return inst[prop] end); if ok2 and v~=nil then p[#p+1]='"'..prop..'":'..j(v) end end end
	for cn,props in pairs(SAFE_PROPS) do if cn~="GuiObject" then local ok2,ic=pcall(function() return inst:IsA(cn) end); if ok2 and ic then for _,prop in ipairs(props) do local ok3,v=pcall(function() return inst[prop] end); if ok3 and v~=nil then p[#p+1]='"'..prop..'":'..j(v) end end end end end
	local ch=inst:GetChildren(); if #ch>0 then local cp={};for _,c in ipairs(ch) do cp[#cp+1]=serInst(c,d+1) end; p[#p+1]='"children":['..table.concat(cp,",").."]" end
	return "{"..table.concat(p,",").."}"
end

local function lex(src)
	local tokens,i,n,line={},1,#src,1
	local function ch(o) return src:sub(i+(o or 0),i+(o or 0)) end
	local function adv(k) i=i+(k or 1) end
	local function push(t,v) tokens[#tokens+1]={t=t,v=v,ln=line} end

	while i<=n do
		local c=ch()
		if c=="\n" then push("NL","");line=line+1;adv()
		elseif c:match("[ \t\r]") then adv()
		elseif c=="-" and ch(1)=="-" then while i<=n and ch()~="\n" do adv() end
		elseif c=="@" then
			adv(); local w=""
			while i<=n and (ch():match("[%a_]") or ch()=="-") do w=w..ch();adv() end
			push("TRG",w:upper())
		elseif c=='"' or c=="'" then
			local q=c;adv();local s=""
			while i<=n and ch()~=q do
				if ch()=="\\" then adv();local e=ch();s=s..(e=="n" and "\n" or e=="t" and "\t" or e)
				else s=s..ch() end;adv()
			end
			if i<=n then adv() end;push("STR",s)
		elseif c:match("%d") then
			local num=""
			while i<=n and ch():match("[%d%.]") do num=num..ch();adv() end
			push("NUM",tonumber(num) or 0)
		elseif c:match("[%a_]") then
			local w=""
			while i<=n and ch():match("[%w_]") do w=w..ch();adv() end
			while ch()=="." and ch(1):match("[%a_]") do
				w=w..".";adv()
				while i<=n and ch():match("[%w_]") do w=w..ch();adv() end
			end
			local up=w:upper()
			local alias=ALIASES[w:lower()]
			if     up=="TRUE"  then push("BOOL",true)
			elseif up=="FALSE" then push("BOOL",false)
			elseif up=="NIL"   then push("NIL_L",nil)
			elseif alias       then push("KW",alias)
			elseif KEYWORDS[up] then push("KW",up)
			else push("ID",w)
			end
		elseif c=="=" and ch(1)=="=" then push("OP","==");adv(2)
		elseif c=="!" and ch(1)=="=" then push("OP","!=");adv(2)
		elseif c=="<" and ch(1)=="=" then push("OP","<=");adv(2)
		elseif c==">" and ch(1)=="=" then push("OP",">=");adv(2)
		elseif c=="=" then push("ASSIGN","=");adv()
		elseif c=="<" then push("OP","<");adv()
		elseif c==">" then push("OP",">");adv()
		elseif c=="+" then push("OP","+");adv()
		elseif c=="-" then push("OP","-");adv()
		elseif c=="*" then push("OP","*");adv()
		elseif c=="/" then push("OP","/");adv()
		elseif c=="%" then push("OP","%");adv()
		elseif c=="^" then push("OP","^");adv()
		elseif c=="(" then push("LP","(");adv()
		elseif c==")" then push("RP",")");adv()
		elseif c=="," then push("COM",",");adv()
		elseif c==":" then push("COL",":");adv()
		elseif c=="." then push("DOT",".");adv()
		elseif c=="{" then push("LBRACE","{");adv()
		elseif c=="}" then push("RBRACE","}");adv()
		else adv()
		end
	end
	push("EOF",""); return tokens
end

local function parse(tokens)
	local pos=1
	local function peek(o) return tokens[pos+(o or 0)] or {t="EOF",v=""} end
	local function consume() local t=tokens[pos];pos=pos+1;return t or {t="EOF",v=""} end
	local function skipNL() while peek().t=="NL" do pos=pos+1 end end
	local function is(tp,v) local t=peek();return t.t==tp and (v==nil or t.v==v) end
	local function eat(tp,v) if is(tp,v) then return consume() end return nil end

	local function readObjRef()
		skipNL()
		if is("LBRACE") then
			consume(); local name=consume().v; eat("RBRACE")
			return {k="objvar",name=name}
		else
			local t=consume()
			return {k="objpath",path=tostring(t.v or "")}
		end
	end

	local parseExpr,parseStmt,parseBlock

	local function parseArgList()
		local args={}
		if is("LP") then
			consume()
			while not is("RP") and not is("EOF") do
				skipNL()
				if is("COM") then consume() end
				if not is("RP") then args[#args+1]=parseExpr() end
			end
			eat("RP")
		end
		return args
	end

	local function parseUIElement(className)
		consume(); local target=readObjRef()
		return {k="uielement",class=className,target=target}
	end

	-- ── Expressions ──────────────────────────────────
	local function parsePrimary()
		skipNL(); local t=peek()

		if     t.t=="NUM"    then consume();return{k="num",v=t.v}
		elseif t.t=="STR"    then consume();return{k="str",v=t.v}
		elseif t.t=="BOOL"   then consume();return{k="bool",v=t.v}
		elseif t.t=="NIL_L"  then consume();return{k="nil"}
		elseif is("KW","LIST")  then consume();return{k="list_new"}
		elseif is("KW","TABLE") then consume();return{k="table_new"}
		elseif is("KW","NOT")   then consume();return{k="unop",op="NOT",x=parsePrimary()}
		elseif is("LP") then consume();local e=parseExpr();eat("RP");return e

			-- Player info
		elseif is("KW","GETUSERNAME")    then consume();return{k="builtin",fn="getusername",   args=parseArgList()}
		elseif is("KW","GETUSERID")      then consume();return{k="builtin",fn="getuserid",     args=parseArgList()}
		elseif is("KW","GETAVATAR")      then consume();return{k="builtin",fn="getavatar",     args=parseArgList()}
		elseif is("KW","GETDISPLAYNAME") then consume();return{k="builtin",fn="getdisplayname",args=parseArgList()}
		elseif is("KW","GETACCOUNTAGE")  then consume();return{k="builtin",fn="getaccountage", args=parseArgList()}

		elseif is("KW","CALL") then
			consume();local name=consume().v
			return{k="call_e",name=name,args=parseArgList()}

			-- list/table accessors
		elseif is("KW","GET")   then consume();local n=consume().v;return{k="list_get",name=n,idx=parseExpr()}
		elseif is("KW","LEN")   then consume();local n=consume().v;return{k="list_len",name=n}
		elseif is("KW","HAS")   then consume();local n=consume().v;return{k="list_has",name=n,val=parseExpr()}
		elseif is("KW","TGET")  then consume();local n=consume().v;return{k="tget_e",name=n,key=parseExpr()}
		elseif is("KW","THAS")  then consume();local n=consume().v;return{k="thas_e",name=n,key=parseExpr()}
		elseif is("KW","TLEN")  then consume();local n=consume().v;return{k="tlen_e",name=n}

		elseif is("KW","GETH") then
			consume();return{k="geth_e",obj=readObjRef()}

			-- 5. VISIBLE obj  → Visible (bool)
		elseif is("KW","VISIBLE") then
			consume();return{k="visible_e",obj=readObjRef()}

			-- 6. COLLIDES obj1 obj2  → AABB overlap (bool)
		elseif is("KW","COLLIDES") then
			consume();local o1=readObjRef();local o2=readObjRef()
			return{k="collides_e",obj1=o1,obj2=o2}

			-- 7. LERP(a, b, t)  → number
		elseif is("KW","LERP") then
			consume();return{k="builtin",fn="lerp",args=parseArgList()}

			-- 8. SIN(degrees) / COS(degrees) / TAN(degrees)
		elseif is("KW","SIN") then
			consume();return{k="builtin",fn="sin",args=parseArgList()}
		elseif is("KW","COS") then
			consume();return{k="builtin",fn="cos",args=parseArgList()}
		elseif is("KW","TAN") then
			consume();return{k="builtin",fn="tan",args=parseArgList()}

			-- 9. BETWEEN(val, min, max) → bool
		elseif is("KW","BETWEEN") then
			consume();return{k="builtin",fn="between",args=parseArgList()}

			-- 10. SIGN(n) / POW(a,b)
		elseif is("KW","SIGN") then
			consume();return{k="builtin",fn="sign",args=parseArgList()}
		elseif is("KW","POW") then
			consume();return{k="builtin",fn="pow",args=parseArgList()}

		elseif t.t=="ID" then
			consume()
			if is("LP") then return{k="call_e",name=t.v,args=parseArgList()} end
			if is("COL") then
				consume();local meth=consume();local margs={}
				while not is("NL") and not is("EOF") and not is("COM") and not is("RP") do
					margs[#margs+1]=parseExpr()
					if is("COM") then consume() end
				end
				return{k="list_me",name=t.v,meth=meth.v:upper(),args=margs}
			end
			return{k="var",name=t.v}
		else
			consume();return{k="nil"}
		end
	end

	local function parsePow()
		local l=parsePrimary()
		while is("OP","^") do consume();l={k="binop",op="^",l=l,r=parsePrimary()} end
		return l
	end
	local function parseUnary()
		if is("OP","-") then consume();return{k="unop",op="-",x=parsePow()} end
		return parsePow()
	end
	local function parseMul()
		local l=parseUnary()
		while is("OP","*") or is("OP","/") or is("OP","%") do
			local op=consume().v;l={k="binop",op=op,l=l,r=parseUnary()}
		end
		return l
	end
	local function parseAdd()
		local l=parseMul()
		while is("OP","+") or is("OP","-") do
			local op=consume().v;l={k="binop",op=op,l=l,r=parseMul()}
		end
		return l
	end
	local function parseCmp()
		local l=parseAdd()
		local CMP={["=="]=1,["!="]=1,["<"]=1,[">"]=1,["<="]=1,[">="]=1}
		while is("OP") and CMP[peek().v] do
			local op=consume().v;l={k="binop",op=op,l=l,r=parseAdd()}
		end
		return l
	end
	local function parseAnd()
		local l=parseCmp()
		while is("KW","AND") do consume();l={k="binop",op="AND",l=l,r=parseCmp()} end
		return l
	end
	parseExpr=function()
		local l=parseAnd()
		while is("KW","OR") do consume();l={k="binop",op="OR",l=l,r=parseAnd()} end
		return l
	end

	parseBlock=function(stopKWs)
		local stmts,stop={},{}
		for _,v in ipairs(stopKWs or {}) do stop[v]=true end
		while true do
			skipNL();local t=peek()
			if t.t=="EOF" then break end
			if t.t=="TRG" then break end
			if t.t=="KW" and stop[t.v] then break end
			local s=parseStmt();if s then stmts[#stmts+1]=s end
		end
		return stmts
	end

	-- ── Statements ───────────────────────────────────
	parseStmt=function()
		skipNL();local t=peek()

		if is("KW","LET") then
			consume();local name=consume().v;eat("ASSIGN")
			return{k="let",name=name,val=parseExpr()}
		elseif is("KW","SET") then
			consume();local name=consume().v;eat("ASSIGN")
			return{k="set_s",name=name,val=parseExpr()}
		elseif is("KW","GLOBAL") then
			consume();local name=consume().v;eat("ASSIGN")
			return{k="global",name=name,val=parseExpr()}
		elseif is("KW","FN") then
			consume();local name=consume().v;local params={}
			if is("LP") then
				consume()
				while not is("RP") and not is("EOF") do
					local pt=consume();if pt.t=="ID" then params[#params+1]=pt.v end
					if is("COM") then consume() end
				end;eat("RP")
			end
			skipNL();local body=parseBlock({"END"});eat("KW")
			return{k="func_def",name=name,params=params,body=body}
		elseif is("KW","CALL") then
			consume();local name=consume().v
			return{k="call_s",name=name,args=parseArgList()}
		elseif is("KW","RETURN") then
			consume()
			if is("NL") or is("EOF") then return{k="return",val={k="nil"}} end
			return{k="return",val=parseExpr()}
		elseif is("KW","BREAK")    then consume();return{k="break"}
		elseif is("KW","CONTINUE") then consume();return{k="continue"}
		elseif is("KW","IF") then
			consume();local cond=parseExpr();skipNL()
			local then_b=parseBlock({"ELSEIF","ELSE","END"})
			local elseifs,else_b={},nil
			while is("KW","ELSEIF") do
				consume();local ec=parseExpr();skipNL()
				elseifs[#elseifs+1]={cond=ec,body=parseBlock({"ELSEIF","ELSE","END"})}
			end
			if is("KW","ELSE") then consume();skipNL();else_b=parseBlock({"END"}) end
			eat("KW")
			return{k="if",cond=cond,then_b=then_b,elseifs=elseifs,else_b=else_b}
		elseif is("KW","FOR") then
			consume()
			local var=consume().v
			if is("ASSIGN") then
				eat("ASSIGN")
				local from=parseExpr();eat("KW","TO");local to=parseExpr()
				local step={k="num",v=1}
				if is("KW","STEP") then consume();step=parseExpr() end
				skipNL();local body=parseBlock({"END"});eat("KW")
				return{k="for",var=var,from=from,to=to,step=step,body=body}
			elseif is("KW","IN") then
				eat("KW","IN")
				if is("KW","CHILDREN") then
					consume();local obj=readObjRef();skipNL()
					local body=parseBlock({"END"});eat("KW")
					return{k="foreach_children",var=var,obj=obj,body=body}
				else
					local listname=consume().v;skipNL()
					local body=parseBlock({"END"});eat("KW")
					return{k="foreach",var=var,list=listname,body=body}
				end
			end
		elseif is("KW","WHILE") then
			consume();local cond=parseExpr();skipNL()
			local body=parseBlock({"END"});eat("KW")
			return{k="while",cond=cond,body=body}
		elseif is("KW","REPEAT") then
			consume();local count=parseExpr()
			if is("KW","TIMES") then consume() end;skipNL()
			local body=parseBlock({"END"});eat("KW")
			return{k="repeat",count=count,body=body}
		elseif is("KW","ADD") then consume();local n=consume().v;return{k="list_add",name=n,val=parseExpr()}
		elseif is("KW","REMOVE") then
			consume();local n=consume().v
			local idx=nil
			if not is("NL") and not is("EOF") then idx=parseExpr() end
			return{k="list_remove",name=n,idx=idx}
		elseif is("KW","TSET") then
			consume();local n=consume().v;local key=parseExpr();local val=parseExpr()
			return{k="tset_s",name=n,key=key,val=val}
		elseif is("KW","TDEL") then
			consume();local n=consume().v;return{k="tdel_s",name=n,key=parseExpr()}
		elseif is("KW","SHOW") then consume();return{k="show",obj=readObjRef()}
		elseif is("KW","HIDE") then consume();return{k="hide",obj=readObjRef()}
		elseif is("KW","TEXT") then
			consume();local obj=readObjRef()
			return{k="uiText",obj=obj,text=parseExpr()}
		elseif is("KW","COLOR") then
			consume();local obj=readObjRef()
			local r=parseExpr();eat("COM");local g=parseExpr();eat("COM")
			return{k="color",obj=obj,r=r,g=g,b=parseExpr()}
		elseif is("KW","TEXTCOLOR") then
			consume();local obj=readObjRef()
			local r=parseExpr();eat("COM");local g=parseExpr();eat("COM")
			return{k="textcolor",obj=obj,r=r,g=g,b=parseExpr()}
		elseif is("KW","ALPHA") then consume();local obj=readObjRef();return{k="alpha",obj=obj,a=parseExpr()}
		elseif is("KW","POS") then
			consume();local obj=readObjRef();local x=parseExpr();eat("COM")
			return{k="pos",obj=obj,x=x,y=parseExpr()}
		elseif is("KW","SIZE") then
			consume();local obj=readObjRef();local w=parseExpr();eat("COM")
			return{k="uiSize",obj=obj,w=w,h=parseExpr()}
		elseif is("KW","ROT") then consume();local obj=readObjRef();return{k="rot",obj=obj,r=parseExpr()}
		elseif is("KW","PROP") then
			consume();local obj=readObjRef();local prop=parseExpr()
			return{k="prop",obj=obj,prop=prop,val=parseExpr()}
		elseif is("KW","TWEEN") then
			consume();local obj=readObjRef()
			local kind="POS"
			if is("KW","POS") or is("KW","SIZE") or is("KW","COLOR") or
				is("KW","ALPHA") or is("KW","ROT") or is("KW","TEXTCOLOR") then
				kind=consume().v
			end
			local args={}
			while not is("NL") and not is("EOF")
				and not is("KW","TIME") and not is("KW","EASE") and not is("KW","THEN") do
				if is("COM") then consume()
				elseif peek().t=="NUM" or peek().t=="ID"
					or is("OP","-") or is("LP") or is("LBRACE") then
					args[#args+1]=parseExpr()
				else break end
			end
			local tnode={k="num",v=0.5}
			if is("KW","TIME") then consume();tnode=parseExpr() end
			local ease="Linear"
			if is("KW","EASE") then
				consume()
				if not is("NL") and not is("EOF") and not is("KW","THEN") then ease=consume().v end
			end
			local thenBody=nil;skipNL()
			if is("KW","THEN") then
				consume();skipNL();thenBody=parseBlock({"END"});eat("KW")
			end
			return{k="tween",obj=obj,kind=kind,args=args,time=tnode,ease=ease,thenBody=thenBody}
		elseif is("KW","WAIT")  then consume();return{k="wait",time=parseExpr()}
		elseif is("KW","FIRE")  then
			consume();local ev=parseExpr()
			local val=nil;if not is("NL") and not is("EOF") then val=parseExpr() end
			return{k="fire",ev=ev,val=val}
		elseif is("KW","LOG") or is("KW","PRINT") then consume();return{k="log",val=parseExpr()}
		elseif is("KW","DESTROY") then consume();return{k="destroy",obj=readObjRef()}
		elseif is("KW","CACHE") then
			consume();local op=consume().v:upper()
			if op=="SET" then local key=parseExpr();local val=parseExpr();return{k="cache_set",key=key,val=val}
			elseif op=="CLEAR"    then return{k="cache_clear",key=parseExpr()}
			elseif op=="CLEARALL" then return{k="cache_clearall"}
			end;return nil
		elseif is("KW","BROADCAST") then
			consume();local ev=parseExpr()
			local val=nil;if not is("NL") and not is("EOF") then val=parseExpr() end
			return{k="broadcast_s",ev=ev,val=val}
		elseif is("KW","PARENT") or is("KW","MOVETO") then
			consume();local obj=readObjRef();local target=readObjRef()
			return{k="parent_s",obj=obj,target=target}
		elseif is("KW","UNPARENT") then consume();return{k="unparent_s",obj=readObjRef()}
		elseif is("KW","WATCH") then
			consume();local varname=consume().v;skipNL()
			local body=parseBlock({"END"});eat("KW")
			return{k="watch_s",varname=varname,body=body}
		elseif is("KW","SCHEDULE") then
			consume();local name=parseExpr()
			local mode="after"
			if is("KW","AFTER") or is("KW","EVERY") then mode=consume().v:lower()
			elseif peek().t=="ID" then mode=consume().v:lower() end
			local delay=parseExpr()
			return{k="schedule_s",name=name,mode=mode,delay=delay}
		elseif is("KW","GBCAST") then
			consume()
			local topic=parseExpr()
			local val=nil
			if not is("NL") and not is("EOF") then val=parseExpr() end
			return{k="gbcast_s",topic=topic,val=val}
		elseif is("KW","SBCAST") then
			consume()
			local topic=parseExpr()
			local val=nil
			if not is("NL") and not is("EOF") then val=parseExpr() end
			return{k="sbcast_s",topic=topic,val=val}
		elseif is("KW","REDIRECT") then
			consume()
			return{k="redirect_s",page=parseExpr()}
		elseif t.t=="ID" and t.v:lower()=="cancel" then
			consume();return{k="cancel_s",name=parseExpr()}
		elseif is("KW","TRANSITION") then consume();return{k="transition_s",state=parseExpr()}
		elseif is("KW","STATE") then
			consume()
			if is("ASSIGN") then consume();return{k="state_set",state=parseExpr()} end
			return nil

			-- ★ NEW STATEMENT COMMANDS 11-20

			-- 11. FLASH obj r, g, b  [time]  → flash color then back
		elseif is("KW","FLASH") then
			consume();local obj=readObjRef()
			local r=parseExpr();eat("COM");local g=parseExpr();eat("COM");local b=parseExpr()
			local tv={k="num",v=0.12}
			if not is("NL") and not is("EOF") then tv=parseExpr() end
			return{k="flash_s",obj=obj,r=r,g=g,b=b,time=tv}

			-- 12. SHAKE obj  [intensity]  → shake position briefly
		elseif is("KW","SHAKE") then
			consume();local obj=readObjRef()
			local intensity={k="num",v=6}
			local duration={k="num",v=0.3}
			if not is("NL") and not is("EOF") then intensity=parseExpr() end
			if not is("NL") and not is("EOF") then duration=parseExpr() end
			return{k="shake_s",obj=obj,intensity=intensity,duration=duration}

			-- 13. SETIMAGE obj  imageId  → set ImageLabel.Image
		elseif is("KW","SETIMAGE") then
			consume();local obj=readObjRef();local id=parseExpr()
			return{k="setimage_s",obj=obj,id=id}

			-- 14. FONT obj  fontName  → set Font enum on TextLabel/TextButton/TextBox
		elseif is("KW","FONT") then
			consume();local obj=readObjRef();local fn=parseExpr()
			return{k="font_s",obj=obj,font=fn}

			-- 15. TEXTSIZE obj  n  → set TextSize
		elseif is("KW","TEXTSIZE") then
			consume();local obj=readObjRef();local n=parseExpr()
			return{k="textsize_s",obj=obj,n=n}

			-- 16. ZINDEX obj  n  → set ZIndex
		elseif is("KW","ZINDEX") then
			consume();local obj=readObjRef();local n=parseExpr()
			return{k="zindex_s",obj=obj,n=n}

			-- 17. ANCHOR obj  ax  ay  → set AnchorPoint (0-1 each)
		elseif is("KW","ANCHOR") then
			consume();local obj=readObjRef()
			local ax=parseExpr();eat("COM");local ay=parseExpr()
			return{k="anchor_s",obj=obj,ax=ax,ay=ay}

			-- 18. CLONETO  src  target  → clone into specific parent
		elseif is("KW","CLONETO") then
			consume();local src=readObjRef();local target=readObjRef()
			return{k="cloneto_s",src=src,target=target}

			-- 19. STOPTWEEN  obj  → cancel active tweens on object
		elseif is("KW","STOPTWEEN") then
			consume();local obj=readObjRef()
			return{k="stoptween_s",obj=obj}

			-- 20. WRAP obj  bool  → set TextWrapped
		elseif is("KW","WRAP") then
			consume();local obj=readObjRef();local b=parseExpr()
			return{k="wrap_s",obj=obj,b=b}

		elseif t.t=="ID" then
			consume()
			if is("LP") then return{k="call_s",name=t.v,args=parseArgList()} end
			if is("COL") then
				consume();local meth=consume();local args={}
				while not is("NL") and not is("EOF") do
					if is("COM") then consume()
					elseif not is("NL") and not is("EOF") then args[#args+1]=parseExpr() end
				end
				return{k="list_ms",name=t.v,meth=meth.v:upper(),args=args}
			end
			if is("ASSIGN") then
				consume();return{k="assign",name=t.v,val=parseExpr()}
			end
			return nil
		else
			consume();return nil
		end
	end

	skipNL()
	local page="all"
	if is("KW","PAGE") then
		consume();local pt=eat("STR") or eat("ID");if pt then page=pt.v end
	end

	local triggers,funcs,glob_stmts={},{},{}

	while not is("EOF") do
		skipNL();if is("EOF") then break end
		if is("KW","FN") then
			local f=parseStmt();if f then funcs[f.name]=f end
		elseif peek().t=="TRG" then
			local ttype=peek().v;consume()
			if ttype~="END" then
				local targ=""
				local nt=peek()
				if nt.t=="STR" or nt.t=="ID" or nt.t=="NUM" then targ=tostring(consume().v) end
				skipNL()
				local body=parseBlock({})
				if peek().t=="TRG" and peek().v=="END" then consume() end
				triggers[#triggers+1]={ttype=ttype,targ=targ,body=body}
			end
		elseif peek().t~="NL" then
			local s=parseStmt();if s then glob_stmts[#glob_stmts+1]=s end
		else
			consume()
		end
	end

	return{page=page,triggers=triggers,funcs=funcs,glob_stmts=glob_stmts}
end

	local s=tostring(math.floor(n or 0));local res,cnt="",0
	for i=#s,1,-1 do cnt=cnt+1;res=s:sub(i,i)..res;if cnt%3==0 and i~=1 then res="."..res end end
	return res
end

	-- 7. LERP(a, b, t) – Linear interpolation, t clamped 0-1
	lerp = function(a,b,t)
		a=a or 0;b=b or 0;t=math.clamp(t or 0,0,1)
		return a+(b-a)*t
	end,

	-- 8. SIN/COS/TAN (degrees → converts internally)
	sin = function(deg) return math.sin(math.rad(deg or 0)) end,
	cos = function(deg) return math.cos(math.rad(deg or 0)) end,
	tan = function(deg) return math.tan(math.rad(deg or 0)) end,

	-- 9. BETWEEN(val, min, max) → bool
	between = function(v,mn,mx) v=v or 0;mn=mn or 0;mx=mx or 0;return v>=mn and v<=mx end,

	-- 10. SIGN / POW
	sign = function(v) v=v or 0;return v>0 and 1 or (v<0 and -1 or 0) end,
	pow  = function(a,b) return (a or 0)^(b or 1) end,
}

-- ★ Helper: register and track an active tween so STOPTWEEN can cancel it
local function trackTween(obj, tw)
	if not obj then return end
	local key = tostring(obj)
	if not _activeTweens[key] then _activeTweens[key]={} end
	table.insert(_activeTweens[key], tw)
	-- Auto-remove when completed
	tw.Completed:Connect(function()
		local list = _activeTweens[key]
		if list then
			for i,v in ipairs(list) do
				if v==tw then table.remove(list,i);break end
			end
		end
	end)
end

local function runScript(ast)
	local stopped,conns,broadcastEntries,scheduledTasks,watchHandlers,currentState
		=false,{},{},{},{},"",""
	local G=newEnv(nil)
	for name,fn in pairs(ast.funcs) do envLet(G,name,fn) end

	local function resolveObj(ref,env)
		if not ref then return nil end
		if ref.k=="objvar" then
			local v=envGet(env,ref.name)
			if typeof(v)=="Instance" then return v end
			if type(v)=="string" then return resolve(v) end
			return nil
		elseif ref.k=="objpath" then
			return resolve(ref.path)
		end
	end

	local function fireWatch(name,old,new_val)
		for _,wh in ipairs(watchHandlers) do
			if wh.varname==name then
				task.spawn(function()
					local e=newEnv(G);envLet(e,"oldValue",old);envLet(e,"newValue",new_val)
					execBlock(wh.body,e)
				end)
			end
		end
	end

	local eval,exec,execBlock

	eval=function(node,env)
		if not node then return nil end
		local k=node.k
		if k=="num"       then return node.v
		elseif k=="bool"  then return node.v
		elseif k=="nil"   then return nil
		elseif k=="list_new"  then return newList()
		elseif k=="table_new" then return newTable_FS()
		elseif k=="var"       then return envGet(env,node.name)
		elseif k=="state_get" then return currentState
		elseif k=="str" then
			return(node.v:gsub("{([%w_]+)}",function(vn)
				local v=envGet(env,vn);return v~=nil and tostring(v) or "{"..vn.."}"
			end))
		elseif k=="unop" then
			if node.op=="-" then return -(eval(node.x,env) or 0) end
			if node.op=="NOT" then return not eval(node.x,env) end
		elseif k=="binop" then
			local op=node.op
			if op=="AND" then local l=eval(node.l,env);if not l then return l end;return eval(node.r,env) end
			if op=="OR"  then local l=eval(node.l,env);if l then return l end;return eval(node.r,env) end
			local l=eval(node.l,env);local r=eval(node.r,env)
			if op=="+" then
				if type(l)=="string" or type(r)=="string" then return tostring(l or "")..tostring(r or "") end
				return (l or 0)+(r or 0)
			elseif op=="-" then return(l or 0)-(r or 0)
			elseif op=="*" then return(l or 0)*(r or 0)
			elseif op=="/" then return(r or 0)~=0 and(l or 0)/r or 0
			elseif op=="%" then return(r or 0)~=0 and(l or 0)%r or 0
			elseif op=="^" then return(l or 0)^(r or 0)
			elseif op=="==" then return l==r
			elseif op=="!=" then return l~=r
			elseif op=="<"  then return(l or 0)<(r or 0)
			elseif op==">"  then return(l or 0)>(r or 0)
			elseif op=="<=" then return(l or 0)<=(r or 0)
			elseif op==">=" then return(l or 0)>=(r or 0)
			end
		elseif k=="builtin" then
			local fn=BUILTINS[node.fn];if not fn then return nil end
			local args={};for _,a in ipairs(node.args) do args[#args+1]=eval(a,env) end
			local ok,result=pcall(fn,table.unpack(args));return ok and result or nil
		elseif k=="call_e" then
			local fn=envGet(env,node.name);if not fn then return nil end
			local args={};for _,a in ipairs(node.args) do args[#args+1]=eval(a,env) end
			if type(fn)=="function" then return fn(table.unpack(args))
			elseif type(fn)=="table" and fn.k=="func_def" then
				local ce=newEnv(G)
				for i,p in ipairs(fn.params) do envLet(ce,p,args[i]) end
				local sig=execBlock(fn.body,ce);if isRet(sig) then return sig.v end
			end
		elseif k=="list_get" then
			local lst=envGet(env,node.name);if lst and lst.__list then return lst.d[math.floor(eval(node.idx,env) or 1)] end
		elseif k=="list_len" then
			local lst=envGet(env,node.name);if lst and lst.__list then return #lst.d end;return 0
		elseif k=="list_has" then
			local lst=envGet(env,node.name);local val=eval(node.val,env)
			if lst and lst.__list then for _,v in ipairs(lst.d) do if v==val then return true end end end;return false
		elseif k=="list_me" then
			local lst=envGet(env,node.name);if not lst or not lst.__list then return nil end
			local args={};for _,a in ipairs(node.args) do args[#args+1]=eval(a,env) end
			local m=node.meth
			if m=="GET" then return lst.d[math.floor(args[1] or 1)]
			elseif m=="LEN" then return #lst.d
			elseif m=="ADD" then lst.d[#lst.d+1]=args[1]
			elseif m=="REMOVE" then table.remove(lst.d,math.floor(args[1] or #lst.d))
			elseif m=="HAS" then for _,v in ipairs(lst.d) do if v==args[1] then return true end end;return false
			end
		elseif k=="tget_e"  then local tbl=envGet(env,node.name);if tbl and tbl.__fstable then return tbl.d[eval(node.key,env)] end
		elseif k=="thas_e"  then local tbl=envGet(env,node.name);if tbl and tbl.__fstable then return tbl.d[eval(node.key,env)]~=nil end;return false
		elseif k=="tlen_e"  then local tbl=envGet(env,node.name);if tbl and tbl.__fstable then local n=0;for _ in pairs(tbl.d) do n=n+1 end;return n end;return 0
		elseif k=="tkeys_e" then
			local tbl=envGet(env,node.name);local lst=newList()
			if tbl and tbl.__fstable then for key in pairs(tbl.d) do lst.d[#lst.d+1]=key end end;return lst
		elseif k=="clone_e" then
			local src=resolveObj(node.src,env)
			if src then local cl=src:Clone();local sc=getScene();if sc then cl.Parent=sc end;return cl end;return nil
		elseif k=="getprop_e" then
			local obj=resolveObj(node.obj,env);local prop=tostring(eval(node.prop,env) or "")
			if obj and prop~="" then
				local val=obj
				for part in prop:gmatch("[^%.]+") do
					if val==nil then return nil end
					local ok,v=pcall(function() return val[part] end);if not ok then return nil end;val=v
				end;return val
			end;return nil
		elseif k=="uielement" then
			local target=resolveObj(node.target,env)
			if not target then warn("[FlowScript] "..node.class..": not found");return nil end
			local inst=Instance.new(node.class);inst.Parent=target;return inst
		elseif k=="vc_udim"  then
			local a={};for _,x in ipairs(node.args) do a[#a+1]=eval(x,env) end;return UDim.new(a[1] or 0,a[2] or 0)
		elseif k=="vc_udim2" then
			local a={};for _,x in ipairs(node.args) do a[#a+1]=eval(x,env) end;return UDim2.new(a[1] or 0,a[2] or 0,a[3] or 0,a[4] or 0)
		elseif k=="vc_color3" then
			local a={};for _,x in ipairs(node.args) do a[#a+1]=eval(x,env) end;return Color3.fromRGB(a[1] or 255,a[2] or 255,a[3] or 255)
		elseif k=="vc_enum" then
			local a={};for _,x in ipairs(node.args) do a[#a+1]=eval(x,env) end
			local ok,r=pcall(function() return Enum[tostring(a[1] or "")][tostring(a[2] or "")] end);return ok and r or nil
		elseif k=="vc_vec2" then
			local a={};for _,x in ipairs(node.args) do a[#a+1]=eval(x,env) end;return Vector2.new(a[1] or 0,a[2] or 0)
		elseif k=="vc_sequence" then
			local a={};for _,x in ipairs(node.args) do a[#a+1]=eval(x,env) end
			local kp={};for i=1,#a-1,2 do local ok,p=pcall(NumberSequenceKeypoint.new,a[i] or 0,a[i+1] or 0);if ok then kp[#kp+1]=p end end
			if #kp>=2 then local ok,s=pcall(NumberSequence.new,kp);return ok and s or nil end;return nil
		elseif k=="vc_colorseq" then
			local a={};for _,x in ipairs(node.args) do a[#a+1]=eval(x,env) end
			local kp={};for i=1,#a-3,4 do local col=Color3.fromRGB(a[i+1] or 255,a[i+2] or 255,a[i+3] or 255);local ok,p=pcall(ColorSequenceKeypoint.new,a[i] or 0,col);if ok then kp[#kp+1]=p end end
			if #kp>=2 then local ok,s=pcall(ColorSequence.new,kp);return ok and s or nil end;return nil
		elseif k=="findfirst_e" then
			local obj=resolveObj(node.obj,env);local name=tostring(eval(node.name,env) or "")
			local rec=eval(node.recursive,env) or false
			if obj and name~="" then local ok,r=pcall(function() return obj:FindFirstChild(name,rec) end);return ok and r or nil end;return nil
		elseif k=="findall_e" then
			local obj=resolveObj(node.obj,env);local cn=tostring(eval(node.classOrName,env) or "");local lst=newList()
			if obj and cn~="" then local ok,d=pcall(function() return obj:GetDescendants() end);if ok then for _,c in ipairs(d) do if c.ClassName==cn or c.Name==cn then lst.d[#lst.d+1]=c end end end end;return lst
		elseif k=="ischildof_e" then
			local c=resolveObj(node.child,env);local p=resolveObj(node.parent,env)
			if c and p then local ok,r=pcall(function() return c:IsDescendantOf(p) end);return ok and r or false end;return false
		elseif k=="getparent_e" then
			local obj=resolveObj(node.obj,env);if obj then local ok,p=pcall(function() return obj.Parent end);return ok and p or nil end;return nil
		elseif k=="getancestor_e" then
			local obj=resolveObj(node.obj,env);local cl=tostring(eval(node.class,env) or "")
			if obj and cl~="" then local ok,r=pcall(function() return obj:FindFirstAncestorOfClass(cl) end);return ok and r or nil end;return nil
		elseif k=="countchildren_e" then
			local obj=resolveObj(node.obj,env);if obj then local ok,c=pcall(function() return #obj:GetChildren() end);return ok and c or 0 end;return 0
		elseif k=="cache_get_e" then return CACHE[tostring(eval(node.key,env) or "")]
		elseif k=="cache_has_e" then return CACHE[tostring(eval(node.key,env) or "")]~=nil

			-- ★ NEW EXPRESSION EVALUATIONS

			-- 1. GETX
		elseif k=="getx_e" then
			local obj=resolveObj(node.obj,env)
			if obj then local ok,v=pcall(function() return obj.Position.X.Offset end);return ok and v or 0 end;return 0

			-- 2. GETY
		elseif k=="gety_e" then
			local obj=resolveObj(node.obj,env)
			if obj then local ok,v=pcall(function() return obj.Position.Y.Offset end);return ok and v or 0 end;return 0

			-- 3. GETW
		elseif k=="getw_e" then
			local obj=resolveObj(node.obj,env)
			if obj then local ok,v=pcall(function() return obj.Size.X.Offset end);return ok and v or 0 end;return 0

			-- 4. GETH
		elseif k=="geth_e" then
			local obj=resolveObj(node.obj,env)
			if obj then local ok,v=pcall(function() return obj.Size.Y.Offset end);return ok and v or 0 end;return 0

			-- 5. VISIBLE
		elseif k=="visible_e" then
			local obj=resolveObj(node.obj,env)
			if obj then local ok,v=pcall(function() return obj.Visible end);return ok and v or false end;return false

			-- 6. COLLIDES (AABB)
		elseif k=="collides_e" then
			local o1=resolveObj(node.obj1,env)
			local o2=resolveObj(node.obj2,env)
			if not o1 or not o2 then return false end
			local ok1,p1=pcall(function() return o1.AbsolutePosition end)
			local ok2,p2=pcall(function() return o2.AbsolutePosition end)
			local ok3,s1=pcall(function() return o1.AbsoluteSize end)
			local ok4,s2=pcall(function() return o2.AbsoluteSize end)
			if not(ok1 and ok2 and ok3 and ok4) then return false end
			local overlapX = p1.X < p2.X + s2.X and p1.X + s1.X > p2.X
			local overlapY = p1.Y < p2.Y + s2.Y and p1.Y + s1.Y > p2.Y
			return overlapX and overlapY
		end
		return nil
	end

	exec=function(stmt,env)
		if stopped or not stmt then return nil end
		local k=stmt.k

		if k=="let" then envLet(env,stmt.name,eval(stmt.val,env))
		elseif k=="set_s" then
			local val=eval(stmt.val,env);local old=envGet(env,stmt.name)
			envSet(env,stmt.name,val);if old~=val then fireWatch(stmt.name,old,val) end
		elseif k=="assign" then
			local val=eval(stmt.val,env);local old=envGet(env,stmt.name)
			envSet(env,stmt.name,val);if old~=val then fireWatch(stmt.name,old,val) end
		elseif k=="global" then
			local val=eval(stmt.val,env);local old=envGet(G,stmt.name)
			envGlobal(env,stmt.name,val);if old~=val then fireWatch(stmt.name,old,val) end
		elseif k=="func_def" then envLet(env,stmt.name,stmt)
		elseif k=="call_s" then
			local fn=envGet(env,stmt.name);if not fn then return nil end
			local args={};for _,a in ipairs(stmt.args) do args[#args+1]=eval(a,env) end
			if type(fn)=="function" then fn(table.unpack(args))
			elseif type(fn)=="table" and fn.k=="func_def" then
				local ce=newEnv(G);for i,p in ipairs(fn.params) do envLet(ce,p,args[i]) end;execBlock(fn.body,ce)
			end
		elseif k=="return"   then return SIG_RETURN(eval(stmt.val,env))
		elseif k=="break"    then return SIG_BREAK
		elseif k=="continue" then return SIG_CONTINUE
		elseif k=="if" then
			if eval(stmt.cond,env) then return execBlock(stmt.then_b,newEnv(env)) end
			for _,ei in ipairs(stmt.elseifs) do if eval(ei.cond,env) then return execBlock(ei.body,newEnv(env)) end end
			if stmt.else_b then return execBlock(stmt.else_b,newEnv(env)) end
		elseif k=="for" then
			local from=eval(stmt.from,env);local to=eval(stmt.to,env);local step=eval(stmt.step,env);local i=from
			while (step>0 and i<=to) or (step<0 and i>=to) do
				if stopped then break end
				local le=newEnv(env);envLet(le,stmt.var,i)
				local sig=execBlock(stmt.body,le);if sig==SIG_BREAK then break end;if isRet(sig) then return sig end
				i=i+step
			end
		elseif k=="foreach" then
			local lst=envGet(env,stmt.list)
			if lst and lst.__list then
				for _,v in ipairs(lst.d) do
					if stopped then break end
					local le=newEnv(env);envLet(le,stmt.var,v)
					local sig=execBlock(stmt.body,le);if sig==SIG_BREAK then break end;if isRet(sig) then return sig end
				end
			end
		elseif k=="foreach_children" then
			local obj=resolveObj(stmt.obj,env)
			if obj then
				for _,child in ipairs(obj:GetChildren()) do
					if stopped then break end
					local le=newEnv(env);envLet(le,stmt.var,child)
					local sig=execBlock(stmt.body,le);if sig==SIG_BREAK then break end;if isRet(sig) then return sig end
				end
			end
		elseif k=="while" then
			local g=0
			while eval(stmt.cond,env) do
				if stopped then break end;g=g+1;if g>200000 then warn("[FS] Infinite loop");break end
				local sig=execBlock(stmt.body,newEnv(env));if sig==SIG_BREAK then break end;if isRet(sig) then return sig end
			end
		elseif k=="repeat" then
			local n=math.floor(eval(stmt.count,env) or 0)
			for _=1,n do
				if stopped then break end
				local sig=execBlock(stmt.body,newEnv(env));if sig==SIG_BREAK then break end;if isRet(sig) then return sig end
			end
		elseif k=="list_add" then local lst=envGet(env,stmt.name);if lst and lst.__list then lst.d[#lst.d+1]=eval(stmt.val,env) end
		elseif k=="list_remove" then
			local lst=envGet(env,stmt.name)
			if lst and lst.__list then local idx=stmt.idx and math.floor(eval(stmt.idx,env) or #lst.d) or #lst.d;table.remove(lst.d,idx) end
		elseif k=="list_ms" then
			local lst=envGet(env,stmt.name)
			if lst and lst.__list then
				local args={};for _,a in ipairs(stmt.args) do args[#args+1]=eval(a,env) end
				local m=stmt.meth
				if m=="ADD" then lst.d[#lst.d+1]=args[1]
				elseif m=="REMOVE" then table.remove(lst.d,math.floor(args[1] or #lst.d))
				elseif m=="SET" then lst.d[math.floor(args[1] or 1)]=args[2]
				end
			end
		elseif k=="tset_s" then local tbl=envGet(env,stmt.name);if tbl and tbl.__fstable then tbl.d[eval(stmt.key,env)]=eval(stmt.val,env) end
		elseif k=="tdel_s" then local tbl=envGet(env,stmt.name);if tbl and tbl.__fstable then tbl.d[eval(stmt.key,env)]=nil end
		elseif k=="show" then local obj=resolveObj(stmt.obj,env);if obj then obj.Visible=true end
		elseif k=="hide" then local obj=resolveObj(stmt.obj,env);if obj then obj.Visible=false end
		elseif k=="uiText" then
			local obj=resolveObj(stmt.obj,env)
			if obj and(obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox")) then obj.Text=tostring(eval(stmt.text,env) or "") end
		elseif k=="color" then
			local obj=resolveObj(stmt.obj,env)
			if obj then obj.BackgroundColor3=Color3.fromRGB(eval(stmt.r,env) or 255,eval(stmt.g,env) or 255,eval(stmt.b,env) or 255) end
		elseif k=="textcolor" then
			local obj=resolveObj(stmt.obj,env)
			if obj then pcall(function() obj.TextColor3=Color3.fromRGB(eval(stmt.r,env) or 255,eval(stmt.g,env) or 255,eval(stmt.b,env) or 255) end) end
		elseif k=="alpha" then local obj=resolveObj(stmt.obj,env);if obj then obj.BackgroundTransparency=1-math.clamp(eval(stmt.a,env) or 1,0,1) end
		elseif k=="pos" then
			local obj=resolveObj(stmt.obj,env)
			if obj then
				local x=eval(stmt.x,env) or 0;local y=eval(stmt.y,env) or 0
				local scaledX,scaledY=DomainUtils.scaleIfWebsite(obj,x,y)
				obj.Position=UDim2.new(0,scaledX,0,scaledY)
			end
		elseif k=="uiSize" then
			local obj=resolveObj(stmt.obj,env)
			if obj then
				local w=eval(stmt.w,env) or 0;local h=eval(stmt.h,env) or 0
				local scaledW,scaledH=DomainUtils.scaleIfWebsite(obj,w,h)
				obj.Size=UDim2.new(0,scaledW,0,scaledH)
			end
		elseif k=="rot" then local obj=resolveObj(stmt.obj,env);if obj then obj.Rotation=eval(stmt.r,env) or 0 end
		elseif k=="prop" then
			local obj=resolveObj(stmt.obj,env);local prop=tostring(eval(stmt.prop,env) or "");local val=eval(stmt.val,env)
			if obj and prop~="" then
				if (prop=="Position" or prop=="Size") and typeof(val)=="UDim2" and DomainUtils.isWebsiteScene(obj) then
					val=DomainUtils.scaleUDim2(val)
				end
				local ok,err=pcall(function() obj[prop]=val end);if not ok then warn("[FS] prop '"..prop.."': "..tostring(err)) end
			end
		elseif k=="tween" then
			local obj=resolveObj(stmt.obj,env);if not obj then return nil end
			local tv=math.max(0.01,eval(stmt.time,env) or 0.5)
			local info=TweenInfo.new(tv,EASE_MAP[stmt.ease] or Enum.EasingStyle.Linear,Enum.EasingDirection.Out)
			local args={};for _,a in ipairs(stmt.args) do args[#args+1]=eval(a,env) end
			local goal={};local kind=stmt.kind
			if kind=="POS" then
				local x,y=args[1] or 0,args[2] or 0
				local sx,sy=DomainUtils.scaleIfWebsite(obj,x,y)
				goal.Position=UDim2.new(0,sx,0,sy)
			elseif kind=="SIZE" then
				local w,h=args[1] or 0,args[2] or 0
				local sw,sh=DomainUtils.scaleIfWebsite(obj,w,h)
				goal.Size=UDim2.new(0,sw,0,sh)
			elseif kind=="COLOR"     then goal.BackgroundColor3=Color3.fromRGB(args[1] or 0,args[2] or 0,args[3] or 0)
			elseif kind=="ALPHA"     then goal.BackgroundTransparency=1-math.clamp(args[1] or 1,0,1)
			elseif kind=="ROT"       then goal.Rotation=args[1] or 0
			elseif kind=="TEXTCOLOR" then pcall(function() goal.TextColor3=Color3.fromRGB(args[1] or 255,args[2] or 255,args[3] or 255) end)
			end
			if next(goal) then
				local tw=TweenService:Create(obj,info,goal);local done=false
				trackTween(obj,tw)   -- ★ track for STOPTWEEN
				local c;c=tw.Completed:Connect(function() done=true;c:Disconnect() end);tw:Play()
				while not done and not stopped do task.wait(0.03) end
			end
			if stmt.thenBody and not stopped then execBlock(stmt.thenBody,newEnv(env)) end
		elseif k=="wait" then
			local total=eval(stmt.time,env) or 0;local el=0
			while el<total and not stopped do local dt=math.min(0.05,total-el);task.wait(dt);el=el+dt end
		elseif k=="fire" then
			local evname=tostring(eval(stmt.ev,env) or "");local evval=stmt.val and eval(stmt.val,env) or nil
			local sc=getScene();if sc then local ev=sc:FindFirstChild(evname,true);if ev and ev:IsA("BindableEvent") then ev:Fire(evval) end end
		elseif k=="log" then logFire(eval(stmt.val,env))
		elseif k=="destroy" then local obj=resolveObj(stmt.obj,env);if obj then pcall(function() obj:Destroy() end) end
		elseif k=="cache_set"      then CACHE[tostring(eval(stmt.key,env) or "")]=eval(stmt.val,env)
		elseif k=="cache_clear"    then CACHE[tostring(eval(stmt.key,env) or "")]=nil
		elseif k=="cache_clearall" then for key in pairs(CACHE) do CACHE[key]=nil end
		elseif k=="broadcast_s" then _broadcastFire(tostring(eval(stmt.ev,env) or ""),stmt.val and eval(stmt.val,env) or nil)
		elseif k=="watch_s"     then watchHandlers[#watchHandlers+1]={varname=stmt.varname,body=stmt.body}
		elseif k=="schedule_s" then
			local name=tostring(eval(stmt.name,env) or "");local delay=eval(stmt.delay,env) or 1;local mode=stmt.mode
			if scheduledTasks[name] then scheduledTasks[name].cancelled=true end
			local entry={cancelled=false};scheduledTasks[name]=entry
			task.spawn(function()
				if mode=="every" then
					while not stopped and not entry.cancelled do task.wait(delay);if not stopped and not entry.cancelled then _broadcastFire(name,nil) end end
				else
					local el=0;while el<delay and not stopped and not entry.cancelled do local dt=math.min(0.05,delay-el);task.wait(dt);el=el+dt end
					if not stopped and not entry.cancelled then _broadcastFire(name,nil) end
				end
			end)
		elseif k=="cancel_s" then
			local name=tostring(eval(stmt.name,env) or "")
			if scheduledTasks[name] then scheduledTasks[name].cancelled=true;scheduledTasks[name]=nil end
		elseif k=="transition_s" then
			local ns=tostring(eval(stmt.state,env) or "");local os=currentState;currentState=ns;_broadcastFire("__state__"..ns,os)
		elseif k=="state_set" then currentState=tostring(eval(stmt.state,env) or "")
		elseif k=="parent_s" then
			local obj=resolveObj(stmt.obj,env);local target=resolveObj(stmt.target,env)
			if obj and target then pcall(function() obj.Parent=target end) end
		elseif k=="gbcast_s" then
			local topic=tostring(eval(stmt.topic,env) or "")
			local val=stmt.val and eval(stmt.val,env) or ""
			task.spawn(function()
				local ok,err=pcall(function() game:GetService("MessagingService"):PublishAsync(topic,tostring(val)) end)
				if not ok then warn("[FS] gbcast error: "..tostring(err)) end
			end)
		elseif k=="sbcast_s" then
			local topic=tostring(eval(stmt.topic,env) or "")
			local val=stmt.val and eval(stmt.val,env) or ""
			task.spawn(function()
				local ok,err=pcall(function()
					local sbEvent=game:GetService("ReplicatedStorage"):FindFirstChild("SBroadcast")
					if sbEvent then sbEvent:FireServer(topic,tostring(val)) end
				end)
				if not ok then warn("[FS] sbcast error: "..tostring(err)) end
			end)
		elseif k=="redirect_s" then
			local page=tostring(eval(stmt.page,env) or "")
			local sc=getScene()
			if sc then
				local ev=sc.Parent and sc.Parent:FindFirstChild("GlobalEvents")
				if not ev then
					local ok2,r=pcall(function() return game.Players.LocalPlayer.PlayerGui.System.UI.GlobalEvents end)
					if ok2 then ev=r end
				end
				if ev then
					local redirectEv=ev:FindFirstChild("RedirectDomain_WebsiteScene")
					if redirectEv and redirectEv:IsA("BindableEvent") then redirectEv:Fire(page)
					else warn("[FS] redirect: RedirectDomain_WebsiteScene not found") end
				else warn("[FS] redirect: GlobalEvents not found") end
			end
		elseif k=="unparent_s" then
			local obj=resolveObj(stmt.obj,env);local sc=getScene()
			if obj and sc then pcall(function() obj.Parent=sc end) end

			-- ★ NEW STATEMENT EXECUTIONS

			-- 11. FLASH obj r g b [time]
			--     Flash background to color then back in half the time each direction.
		elseif k=="flash_s" then
			local obj=resolveObj(stmt.obj,env);if not obj then return nil end
			local r=eval(stmt.r,env) or 255
			local g=eval(stmt.g,env) or 255
			local b=eval(stmt.b,env) or 255
			local tv=math.max(0.04,eval(stmt.time,env) or 0.12)
			-- Save original color
			local ok,origCol=pcall(function() return obj.BackgroundColor3 end)
			if not ok then return nil end
			-- Flash in
			local infoIn=TweenInfo.new(tv,Enum.EasingStyle.Sine,Enum.EasingDirection.Out)
			local twIn=TweenService:Create(obj,infoIn,{BackgroundColor3=Color3.fromRGB(r,g,b)})
			trackTween(obj,twIn);twIn:Play()
			task.wait(tv)
			-- Flash out
			local infoOut=TweenInfo.new(tv*0.8,Enum.EasingStyle.Sine,Enum.EasingDirection.In)
			local twOut=TweenService:Create(obj,infoOut,{BackgroundColor3=origCol})
			trackTween(obj,twOut);twOut:Play()
			task.wait(tv*0.8)

			-- 12. SHAKE obj [intensity] [duration]
			--     Rapid position jitter around current position.
		elseif k=="shake_s" then
			local obj=resolveObj(stmt.obj,env);if not obj then return nil end
			local intensity=eval(stmt.intensity,env) or 6
			local duration=eval(stmt.duration,env) or 0.3
			local ok,origPos=pcall(function() return obj.Position end)
			if not ok then return nil end
			local elapsed=0
			local stepTime=0.03
			while elapsed<duration and not stopped do
				local ox=math.random(-intensity,intensity)
				local oy=math.random(-intensity,intensity)
				pcall(function()
					obj.Position=UDim2.new(
						origPos.X.Scale, origPos.X.Offset+ox,
						origPos.Y.Scale, origPos.Y.Offset+oy
					)
				end)
				task.wait(stepTime)
				elapsed=elapsed+stepTime
			end
			pcall(function() obj.Position=origPos end)

			-- 13. SETIMAGE obj  imageId
			--     Set ImageLabel.Image (supports rbxassetid:// strings)
		elseif k=="setimage_s" then
			local obj=resolveObj(stmt.obj,env);local id=tostring(eval(stmt.id,env) or "")
			if obj then
				pcall(function()
					if obj:IsA("ImageLabel") or obj:IsA("ImageButton") then
						obj.Image=id
					end
				end)
			end

			-- 14. FONT obj  "GothamBold" (or any Enum.Font name)
		elseif k=="font_s" then
			local obj=resolveObj(stmt.obj,env);local fontName=tostring(eval(stmt.font,env) or "Gotham")
			if obj then
				pcall(function()
					local enumVal=Enum.Font[fontName]
					if enumVal then obj.Font=enumVal end
				end)
			end

			-- 15. TEXTSIZE obj  n
		elseif k=="textsize_s" then
			local obj=resolveObj(stmt.obj,env);local n=eval(stmt.n,env) or 14
			if obj then pcall(function() obj.TextSize=math.floor(n) end) end

			-- 16. ZINDEX obj  n
		elseif k=="zindex_s" then
			local obj=resolveObj(stmt.obj,env);local n=eval(stmt.n,env) or 1
			if obj then pcall(function() obj.ZIndex=math.floor(n) end) end

			-- 17. ANCHOR obj  ax  ay   (0–1 for X, 0–1 for Y)
		elseif k=="anchor_s" then
			local obj=resolveObj(stmt.obj,env)
			local ax=math.clamp(eval(stmt.ax,env) or 0,0,1)
			local ay=math.clamp(eval(stmt.ay,env) or 0,0,1)
			if obj then pcall(function() obj.AnchorPoint=Vector2.new(ax,ay) end) end

			-- 18. CLONETO  src  target  → clone src into target parent, returns clone
		elseif k=="cloneto_s" then
			local src=resolveObj(stmt.src,env)
			local target=resolveObj(stmt.target,env)
			if src and target then
				pcall(function()
					local cl=src:Clone()
					cl.Parent=target
				end)
			end

			-- 19. STOPTWEEN  obj  → cancel all active tweens on obj
		elseif k=="stoptween_s" then
			local obj=resolveObj(stmt.obj,env);if not obj then return nil end
			local key=tostring(obj)
			local list=_activeTweens[key]
			if list then
				for _,tw in ipairs(list) do pcall(function() tw:Cancel() end) end
				_activeTweens[key]={}
			end

			-- 20. WRAP obj  bool  → set TextWrapped
		elseif k=="wrap_s" then
			local obj=resolveObj(stmt.obj,env);local b=eval(stmt.b,env)
			if obj then
				pcall(function()
					if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
						obj.TextWrapped=(b==true or b=="true" or b==1)
					end
				end)
			end
		end
		return nil
	end

	execBlock=function(stmts,env)
		for _,s in ipairs(stmts) do
			if stopped then return nil end
			local sig=exec(s,env);if sig~=nil then return sig end
		end
		return nil
	end

	local unloadBodies={}
	local function liveAttach(fn, useRetry)
		local inner=nil
		local retryCount=0
		local maxRetries=useRetry and 30 or 1
		local function doAttach()
			if inner then pcall(function() inner:Disconnect() end);inner=nil end
			local success,result=pcall(fn)
			if success then
				inner=result
				if inner==nil and useRetry and retryCount<maxRetries then
					retryCount=retryCount+1;task.wait(0.1);return doAttach()
				end
			end
			return inner
		end
		if useRetry then
			task.spawn(function() task.wait(0.3);if not stopped then doAttach() end end)
		else doAttach() end
		local sc=PlayScene:GetPropertyChangedSignal("Value"):Connect(function()
			if not stopped then retryCount=0;task.wait(0.3);doAttach() end
		end)
		conns[#conns+1]=sc
		conns[#conns+1]=setmetatable({},{__index={Disconnect=function() if inner then pcall(function() inner:Disconnect() end) end end}})
	end

	local function buildTriggers()
		for _,trig in ipairs(ast.triggers) do
			local tt,targ,body=trig.ttype,trig.targ,trig.body

			if tt=="LOAD" then task.spawn(function() if not stopped then execBlock(body,newEnv(G)) end end)
			elseif tt=="UNLOAD" then unloadBodies[#unloadBodies+1]=body
			elseif tt=="TIMER" then
				local iv=tonumber(targ) or 1
				task.spawn(function() while not stopped do task.wait(iv);if not stopped then execBlock(body,newEnv(G)) end end end)
			elseif tt=="ONCE" then
				task.delay(tonumber(targ) or 0,function() if not stopped then execBlock(body,newEnv(G)) end end)
			elseif tt=="CLICK" then
				liveAttach(function()
					local obj=resolveWithRetry(targ,25,0.1)
					if obj and obj:IsA("GuiButton") then
						return obj.MouseButton1Click:Connect(function() if not stopped then task.spawn(function() execBlock(body,newEnv(G)) end) end end)
					else warn("[FS] @click: '"..targ.."' not found") end;return nil
				end)
			elseif tt=="HOVER" then
				liveAttach(function()
					local obj=resolveWithRetry(targ,25,0.1)
					if obj then return obj.MouseEnter:Connect(function() if not stopped then task.spawn(function() execBlock(body,newEnv(G)) end) end end) end;return nil
				end)
			elseif tt=="LEAVE" then
				liveAttach(function()
					local obj=resolveWithRetry(targ,25,0.1)
					if obj then return obj.MouseLeave:Connect(function() if not stopped then task.spawn(function() execBlock(body,newEnv(G)) end) end end) end;return nil
				end)
			elseif tt=="KEYDOWN" or tt=="KEY" then
				local kn=targ~="" and targ or "Return"
				conns[#conns+1]=UserInputService.InputBegan:Connect(function(inp,gp)
					if gp then return end;local ok,kc=pcall(function() return Enum.KeyCode[kn] end)
					if ok and inp.KeyCode==kc and not stopped then task.spawn(function() execBlock(body,newEnv(G)) end) end
				end)
			elseif tt=="KEYUP" then
				local kn=targ~="" and targ or "Return"
				conns[#conns+1]=UserInputService.InputEnded:Connect(function(inp,gp)
					if gp then return end;local ok,kc=pcall(function() return Enum.KeyCode[kn] end)
					if ok and inp.KeyCode==kc and not stopped then task.spawn(function() execBlock(body,newEnv(G)) end) end
				end)
			elseif tt=="KEYHELD" then
				local kn=targ~="" and targ or "Return";local holding=false
				local c1=UserInputService.InputBegan:Connect(function(inp,gp)
					if gp then return end;local ok,kc=pcall(function() return Enum.KeyCode[kn] end)
					if ok and inp.KeyCode==kc and not holding then holding=true;task.spawn(function() while holding and not stopped do execBlock(body,newEnv(G));task.wait(0.03) end end) end
				end)
				local c2=UserInputService.InputEnded:Connect(function(inp,gp)
					local ok,kc=pcall(function() return Enum.KeyCode[kn] end);if ok and inp.KeyCode==kc then holding=false end
				end)
				conns[#conns+1]=c1;conns[#conns+1]=c2
			elseif tt=="EVENT" then
				liveAttach(function()
					local sc=getScene()
					if sc and targ~="" then
						local ev=nil
						for attempt=1,25 do ev=sc:FindFirstChild(targ,true);if ev and ev:IsA("BindableEvent") then break end;if attempt<25 then task.wait(0.1) end end
						if ev and ev:IsA("BindableEvent") then
							return ev.Event:Connect(function(val)
								if not stopped then local e=newEnv(G);envLet(e,"eventValue",val);task.spawn(function() execBlock(body,e) end) end
							end)
						end
					end;return nil
				end)
			elseif tt=="CHANGED" then
				liveAttach(function()
					local path,prop=targ:match("^(.-)%.([^%.]+)$")
					if not path then path=targ;prop=nil end
					local obj=resolve(path)
					if obj then
						if prop then
							local ok,conn=pcall(function()
								return obj:GetPropertyChangedSignal(prop):Connect(function()
									if not stopped then local e=newEnv(G);pcall(function() envLet(e,"changedValue",obj[prop]) end);task.spawn(function() execBlock(body,e) end) end
								end)
							end);if ok then return conn end
						else
							return obj.Changed:Connect(function(cp)
								if not stopped then local e=newEnv(G);envLet(e,"changedProp",cp);pcall(function() envLet(e,"changedValue",obj[cp]) end);task.spawn(function() execBlock(body,e) end) end
							end)
						end
					end;return nil
				end)
			elseif tt=="BROADCAST" then
				local entry={event=targ,fn=function(val) if not stopped then local e=newEnv(G);envLet(e,"broadcastValue",val);execBlock(body,e) end end}
				_bHandlers[#_bHandlers+1]=entry;broadcastEntries[#broadcastEntries+1]=entry
			elseif tt=="SCHEDULE" then
				local entry={event=targ,fn=function() if not stopped then execBlock(body,newEnv(G)) end end}
				_bHandlers[#_bHandlers+1]=entry;broadcastEntries[#broadcastEntries+1]=entry
			elseif tt=="STATE" then
				local entry={event="__state__"..targ,fn=function(os) if not stopped then local e=newEnv(G);envLet(e,"oldState",os);envLet(e,"newState",targ);execBlock(body,e) end end}
				_bHandlers[#_bHandlers+1]=entry;broadcastEntries[#broadcastEntries+1]=entry
			elseif tt=="GBROADCAST" then
				task.spawn(function()
					local ok,err=pcall(function()
						local conn=game:GetService("MessagingService"):SubscribeAsync(targ,function(msg)
							if not stopped then
								local e=newEnv(G)
								envLet(e,"broadcastValue",tostring(msg.Data or ""))
								envLet(e,"broadcastTime",msg.Sent or 0)
								task.spawn(function() execBlock(body,e) end)
							end
						end)
						conns[#conns+1]=conn
					end)
					if not ok then warn("[FS] @gbroadcast error: "..tostring(err)) end
				end)
		elseif tt=="SBROADCAST" then
			local sbEvent=game:GetService("ReplicatedStorage"):FindFirstChild("SBroadcast")
			if sbEvent then
				local conn=sbEvent.OnClientEvent:Connect(function(topic,msg)
					if not stopped and topic==targ then
						local e=newEnv(G)
						envLet(e,"broadcastValue",tostring(msg or ""))
						envLet(e,"broadcastTopic",tostring(topic))
						task.spawn(function() execBlock(body,e) end)
					end
				end)
				conns[#conns+1]=conn
			else
				warn("[FS] @sbroadcast: SBroadcast RemoteEvent not found")
			end
			elseif tt=="LOG-JSON" then
				task.spawn(function()
					if not stopped then
						local sc=getScene();if not sc then logFire("[LOG-JSON] No scene");return end
						local ok,js=pcall(serInst,sc,0)
						if ok then logFire("[LOG-JSON] "..sc.Name..":\n"..js) else logFire("[LOG-JSON] Error: "..tostring(js)) end
					end
				end)
			end
		end
	end

	local RT={}
	function RT:Start() task.wait(0.4);if stopped then return end;execBlock(ast.glob_stmts,G);buildTriggers() end
	function RT:Stop()
		stopped=true
		for _,e in pairs(scheduledTasks) do e.cancelled=true end;scheduledTasks={}
		for _,e in ipairs(broadcastEntries) do for i,h in ipairs(_bHandlers) do if h==e then table.remove(_bHandlers,i);break end end end
		broadcastEntries={}
		for _,c in ipairs(conns) do pcall(function() c:Disconnect() end) end;conns={}
		for _,b in ipairs(unloadBodies) do pcall(function() execBlock(b,newEnv(G)) end) end
	end
	return RT
end

function FlowScriptParser.CreateRuntime(code)
	local ok1,tokens=pcall(lex,code);if not ok1 then warn("[Parser] Lexer: "..tostring(tokens));return nil end
	local ok2,ast=pcall(parse,tokens);if not ok2 then warn("[Parser] Parser: "..tostring(ast));return nil end
	local ok3,rt=pcall(runScript,ast);if not ok3 then warn("[Parser] Interpreter: "..tostring(rt));return nil end
	return rt
end

FlowScriptParser.Cache=CACHE
return FlowScriptParser
